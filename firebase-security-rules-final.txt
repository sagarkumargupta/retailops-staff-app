# ULTRA-SIMPLE FIREBASE SECURITY RULES
# This version allows all authenticated users to access everything

rules_version = '2';
service cloud.firestore {
  match /databases/{db}/documents {
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}

# FIREBASE STORAGE RULES
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
